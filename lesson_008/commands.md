# Основные команды

```bash
# Базовая перезапись
git rebase <ветка>  # Перебазировать текущую ветку на указанную

# Интерактивная перезапись
git rebase -i HEAD~3    # Редактирование последних 3 коммитов
git rebase -i <commit>  # Редактирование с указанного коммита

# Процесс перезапись
git rebase --continue   # Продолжить перезапись после разрешения конфликта
git rebase --abort  # Отменить rebase
git rebase --skip   # Пропустить текущий коммит

# Проверка состояния
git status  # Проверка состояния
git log --oneline --graph --all # Просмотр графа каждой ветки

# Переименования
git config --global alias.<название> '<команда>' # Создать переименование для определенной команды
```
